function calculateDiscountPrice(price: number, category: Category, quantity: number, coupon?: string): number {
    let discount = 0;
    switch (category) {
        case Category.Electronics:
            discount += 0.10;
            break;
        case Category.Books:
            discount += 0.15;

        case Category.Clothing:

    return totalPrice;
}

exercise 7:

Write a TypeScript function that takes a shape object as its parameter. The shape can be a circle, square, or rectangle. The function should calculate and return the area of the shape.

circle: PI * radius ** 2
square: sideLength ** 2
rectangle: width * height 

type Circle = {
    kind: 'circle';
    radius: number;
};

type Square = {
    kind: 'square';
    sideLength: number;
};

type Rectangle = {
    kind: 'rectangle';
    width: number;
    height: number;
};

type Shape = Circle | Square | Rectangle;


function calculateArea(shape: Shape): number|false {
    if (shape.kind === 'circle') {
        return Math.PI * shape.radius ** 2;
    } else if (shape.kind === 'square') {
        return shape.sideLength ** 2;
    } else if (shape.kind === 'rectangle') {
        return shape.width * shape.height;
    }
  return false
}

const circle: Circle = { kind: 'circle', radius: 5 };
const square: Square = { kind: 'square', sideLength: 10 };
const rectangle: Rectangle = { kind: 'rectangle', width: 4, height: 8 };

console.log(calculateArea(circle));      // Output: 78.53981633974483
console.log(calculateArea(square));      // Output: 100
console.log(calculateArea(rectangle));   // Output: 32

type User = {
    id: string;
    name: string;
    subscription?: {
        status: 'active' | 'inactive';
        plan?: {
            name: 'free' | 'premium';
        };
    };
};

function getSubscriptionMessage(user: User): string {
    const status = user.subscription?.status ?? 'inactive';
    const planName = user.subscription?.plan?.name as 'free' | 'premium' | 'no plan';

    if (status === 'active') {
        switch (planName) {
            case 'premium':
                return `${user.name} is subscribed to the Premium plan.`;
            case 'free':
                return `${user.name} is subscribed to the Free plan.`;
            default:
                return `${user.name} has an active subscription with no specific plan.`;
        }
    } else {
        return `${user.name} does not have an active subscription.`;
    }
}


const activePremiumUser: User = {
    id: '1',
    name: 'Alice',
    subscription: { status: 'active', plan: { name: 'premium' } }
};

const inactiveUser: User = {
    id: '2',
    name: 'Bob'
};

console.log(getSubscriptionMessage(activePremiumUser)); // "Alice is subscribed to the Premium plan."
console.log(getSubscriptionMessage(inactiveUser)); // "Bob does not have an active subscription."

class Book {
    readonly title: string;
    author: string;
    private _pages: number;
    isDigital: boolean;

    constructor(title: string, author: string, pages: number, isDigital: boolean = false) {
        this.title = title;
        this.author = author;
        this._pages = pages;
        this.isDigital = isDigital;
    }

    private formatPages(): string {
        return this.isDigital ? `${this._pages} digital pages` : `${this._pages} pages`;
    }

    public getPages(): string {
        return this.formatPages();
    }

    describe(): string {
        return `${this.title} by ${this.author} with ${this.getPages()}`;
    }
}
